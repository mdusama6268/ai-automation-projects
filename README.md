Day 5 — Conversational AI with Memory
Real chatbots remember the conversation. Here is how you build that.

python
from openai import OpenAI

client = OpenAI(api_key="your_key")

conversation_history = [
    {"role": "system", "content": "You are a helpful assistant for a mechanical engineering company."}
]

def chat(user_message):
    conversation_history.append({"role": "user", "content": user_message})
    
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=conversation_history
    )
    
    assistant_reply = response.choices[0].message.content
    conversation_history.append({"role": "assistant", "content": assistant_reply})
    
    return assistant_reply

# Now it remembers the full conversation
print(chat("My name is Rahul and I work in manufacturing."))
print(chat("What industry did I say I work in?"))  # It will remember!
print(chat("Suggest 3 AI use cases for my industry."))
What is happening here: You are passing the full conversation history every time. The model has no memory — you give it the memory by including past messages.

import PyPDF2

def extract_pdf_text(filename):
    text = ""
    with open(filename, "rb") as file:
        reader = PyPDF2.PdfReader(file)
        total_pages = len(reader.pages)
        print(f"Total Pages: {total_pages}")
        
        for page_num in range(total_pages):
            page = reader.pages[page_num]
            text += f"\n--- Page {page_num + 1} ---\n"
            text += page.extract_text()
    
    return text

# Function 1 — Ask AI
def ask_ai(system_prompt, user_message):
    response = client.chat.completions.create(
        model="llama-3.3-70b-versatile",
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_message}
        ]
    )
    return response.choices[0].message.content

# Function 2 — Read PDF
def read_pdf(filename):
    text = ""
    with open(filename, "rb") as file:
        reader = PyPDF2.PdfReader(file)
        print(f"PDF loaded — {len(reader.pages)} pages found")
        for page in reader.pages:
            text += page.extract_text()
    return text

# Function 3 — Get Weather
def get_weather(city):
    api_key = "bd5e378503939ddaee76f12ad7a97608"
    url = f"https://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units=metric"
    data = requests.get(url).json()
    return {
        "city": data["name"],
        "temp": data["main"]["temp"],
        "condition": data["weather"][0]["description"],
        "humidity": data["main"]["humidity"],
        "wind": data["wind"]["speed"]
    }

# Function 4 — Classify Complaint
def classify_complaint(complaint):
    result = ask_ai(
        system_prompt="You are a complaint classifier. Return ONLY one word: DELAY, WRONG_ORDER, REFUND, or COMPLAINT",
        user_message=complaint
    )
    return result.strip()

complaints = [

# Route each complaint
routing_results = []

for complaint in complaints:
    category = classify_complaint(complaint)
    
    # Assign team and action
    if "DELAY" in category:
        team = "Logistics Team"
        action = "Check driver GPS location"
        priority = "🔴 HIGH"
    elif "WRONG_ORDER" in category:
        team = "Restaurant Team"
        action = "Verify order with restaurant"
        priority = "🟡 MEDIUM"
    elif "REFUND" in category:
        team = "Finance Team"
        action = "Process refund in 24 hours"
        priority = "🟡 MEDIUM"
    elif "COMPLAINT" in category:
        team = "Quality Team"
        action = "Escalate to quality manager"
        priority = "🔴 HIGH"
    else:
        team = "Support Team"
        action = "Review manually"
        priority = "🟢 LOW"
    
    routing_results.append({
        "complaint": complaint,
        "category": category,
        "team": team,
        "action": action,
        "priority": priority
    })
    
    print(f"{priority} | {category}")
    print(f"Complaint : {complaint}")
    print(f"Assigned  : {team}")
    print(f"Action    : {action}")
    print("-" * 55)

print(f"\nTotal complaints processed: {len(complaints)}")

    }
]

def command_center_chat(message):
    command_center_history.append({
        "role": "user",
        "content": message
    })
    
    response = client.chat.completions.create(
        model="llama-3.3-70b-versatile",
        messages=command_center_history
    )
    
    reply = response.choices[0].message.content
    
    command_center_history.append({
        "role": "assistant",
        "content": reply
    })
    
    return reply
